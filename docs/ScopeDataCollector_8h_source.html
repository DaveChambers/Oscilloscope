58 state = State::collecting;

60 }
61 }
62
63 if (state == State::collecting)
64 {
65 while (index++ < numSamples)
66 {
67 buffer[numCollected++] = *data++;
68
69 if (numCollected == buffer.size())
70 {
71 audioBufferQueue.push(buffer.data(), buffer.size());
72 state = State::waitingForTrigger;
73 prevSample = SampleType(100);
74 std::fill(buffer.begin(), buffer.end(), 0);
75 if (index < numSamples)
76 {
77 numCollected = numSamples - index;
78 std::copy(data, data + numCollected, buffer.begin());
79 }
80

82 }
83 }
84 }
