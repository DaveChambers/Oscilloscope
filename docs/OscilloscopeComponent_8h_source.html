111 void paint(juce::Graphics& g) override
112 {
113 g.fillAll(juce::Colours::black);

115
116 auto area = getLocalBounds();
117 auto h = (SampleType)area.getHeight();
118 auto w = (SampleType)area.getWidth();
119
120 if (gridCheck)

125 // Oscilloscope

173 static void plot(const SampleType* data,
174 size_t numSamples,
175 juce::Graphics& g,
176 juce::Rectangle<SampleType> rect,
177 SampleType scaler = SampleType(1),
178 SampleType offset = SampleType(0))
179 {
180 auto w = rect.getWidth();
181 auto h = rect.getHeight();
182 auto right = rect.getRight();
183
184 auto center = rect.getBottom() - offset;
185 auto gain = h * scaler;
186
187 g.setColour(juce::Colours::beige);
188
189 for (size_t i = 1; i < numSamples; ++i)
190 g.drawLine({ juce::jmap(SampleType(i - 1), SampleType(0), SampleType(numSamples - 1), SampleType(right - w), SampleType(right)),
191 center - gain * data[i - 1],
192 juce::jmap(SampleType(i), SampleType(0), SampleType(numSamples - 1), SampleType(right - w), SampleType(right)),
193 center - gain * data[i] });
194 }
