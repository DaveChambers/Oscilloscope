72 void drawGrid(juce::Graphics& g, SampleType w, SampleType h)
73 {
74 g.drawLine(0, h / 2, w, h / 2);
75 g.drawLine(1, 0, 1, h);
76
77 for (size_t i = 0; i < 10; i++)
78 {
79 float xPos = i * w / 10;
80 float yPos = i * h / 10;
81 g.drawLine(xPos, h / 2 - 5, xPos, h / 2 + 5);
82 g.drawLine(0, yPos, 5, yPos);
83 }
84
85 float fontHeight = g.getCurrentFont().getAscent();
86 float duration = sampleData.size() * 1000 / sampleRate;
87 duration /= 10;
88 auto xText = juce::String(duration);

91 g.drawLine(w - 65, h - 39, w - 65, h - 39 - fontHeight);

93 g.drawLine(w - 55, h - 39, w - 55, h - 39 - fontHeight);
94 g.drawSingleLineText(xText, w - 50, h - 39);
95

108 void paint(juce::Graphics& g) override
109 {
110 g.fillAll(juce::Colours::black);
111 g.setColour(juce::Colours::white);
112
113 auto area = getLocalBounds();
114 auto h = (SampleType)area.getHeight();
115 auto w = (SampleType)area.getWidth();
116
117 if (gridCheck)

119 drawGrid(g, w, h);

121
122 // Oscilloscope
123 auto scopeRect = juce::Rectangle<SampleType>{ SampleType(0), SampleType(0), w, h };
124 plot(sampleData.data(), sampleData.size(), g, scopeRect, SampleType(1), h / 2);
125
126 g.setColour(juce::Colours::dimgrey);

170 static void plot(const SampleType* data,
171 size_t numSamples,
172 juce::Graphics& g,
173 juce::Rectangle<SampleType> rect,
174 SampleType scaler = SampleType(1),
175 SampleType offset = SampleType(0))
176 {
177 auto w = rect.getWidth();
178 auto h = rect.getHeight();
179 auto right = rect.getRight();
180
181 auto center = rect.getBottom() - offset;
182 auto gain = h * scaler;
183
184 g.setColour(juce::Colours::beige);
185
186 for (size_t i = 1; i < numSamples; ++i)
187 g.drawLine({ juce::jmap(SampleType(i - 1), SampleType(0), SampleType(numSamples - 1), SampleType(right - w), SampleType(right)),
188 center - gain * data[i - 1],
189 juce::jmap(SampleType(i), SampleType(0), SampleType(numSamples - 1), SampleType(right - w), SampleType(right)),
190 center - gain * data[i] });
191 }
