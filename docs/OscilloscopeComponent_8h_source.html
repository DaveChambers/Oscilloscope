72 void drawGrid(juce::Graphics& g, float w, float h)
73 {
74 g.setColour(juce::Colours::ghostwhite);
75 g.setOpacity(0.8);
76 g.drawLine(0, h / 2, w, h / 2);
77 g.drawLine(1, 0, 1, h);
78
79 for (size_t i = 0; i < 10; i++)
80 {
81 float xPos = i * w / 10;
82 float yPos = i * h / 10;
83 g.drawLine(xPos, h / 2 - 8, xPos, h / 2 + 8);
84 g.drawLine(0, yPos, 8, yPos);
85 }
86

88 float duration = sampleData.size() * 1000 / sampleRate;
89 duration /= 10;

93 g.drawLine(w - 65, h - 39, w - 65, h - 39 - fontHeight);
94 g.drawLine(w - 55, h - 39 - fontHeight / 2, w - 65, h - 39 - fontHeight / 2);
95 g.drawLine(w - 55, h - 39, w - 55, h - 39 - fontHeight);
96 g.drawSingleLineText(xText, w - 50, h - 39);
97
98 auto yText = "0.1";
99 g.drawLine(w - 65, h - 19, w - 55, h - 19);
100 g.drawLine(w - 60, h - 19 - fontHeight, w - 60, h - 19);
101 g.drawLine(w - 65, h - 19 - fontHeight, w - 55, h - 19 - fontHeight);
102 g.drawSingleLineText(yText, w - 50, h - 19);

110 void paint(juce::Graphics& g) override
111 {
112 g.fillAll(juce::Colours::black);
113 g.setColour(juce::Colours::white);
114
115 auto area = getLocalBounds();
116 auto h = (float)area.getHeight();
117 auto w = (float)area.getWidth();
118
119 if (gridCheck)
120 {
121 drawGrid(g, w, h);
122 }
123
124 // Oscilloscope
125 auto scopeRect = juce::Rectangle<float>{ float(0), float(0), w, h };
126 plot(sampleData.data(), sampleData.size(), g, scopeRect, float(1), h / 2);
127
128 g.setColour(juce::Colours::dimgrey);
129 auto contour = juce::Line<float>(0, h, w, h);

174 static void plot(const float* data,
175 size_t numSamples,
176 juce::Graphics& g,
177 juce::Rectangle<float> rect,
178 float scaler = float(1),
179 float offset = float(0))
180 {
181 auto w = rect.getWidth();
182 auto h = rect.getHeight();
183 auto right = rect.getRight();
184
185 auto center = rect.getBottom() - offset;
186 auto gain = h * scaler;
187
188 g.setColour(juce::Colours::beige);
189
190 for (size_t i = 1; i < numSamples; ++i)
191 g.drawLine({ juce::jmap(float(i - 1), float(0), float(numSamples - 1), float(right - w), float(right)),
192 center - gain * data[i - 1],
193 juce::jmap(float(i), float(0), float(numSamples - 1), float(right - w), float(right)),
194 center - gain * data[i] });
195 }
