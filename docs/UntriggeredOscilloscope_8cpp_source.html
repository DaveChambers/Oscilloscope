18void UntriggeredOscilloscope::plot(juce::Graphics& g, juce::Rectangle<float> rect, float scaler, float offset)
19{
20 auto data = sampleData.data();
21 auto numSamples = sampleData.size();
22
23 auto w = rect.getWidth();
24 auto h = rect.getHeight();
25 auto right = rect.getRight();
26
27 auto center = rect.getBottom() - offset;
28 auto gain = h * scaler;
29
30 g.setColour(WAVEFORMCOLOUR());
31
32 for (size_t i = 1; i < numSamples; ++i)
33 {
34 g.drawLine({ juce::jmap(float(i - 1), float(0), float(numSamples - 1), float(right - w), float(right)),
35 center - gain * data[i - 1],
36 juce::jmap(float(i), float(0), float(numSamples - 1), float(right - w), float(right)),
37 center - gain * data[i] });
38 }
39}
