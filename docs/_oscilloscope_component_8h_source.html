110 static void plot(const SampleType* data,
111 size_t numSamples,
112 juce::Graphics& g,
113 juce::Rectangle<SampleType> rect,
114 SampleType scaler = SampleType(1),
115 SampleType offset = SampleType(0))
116 {
117 auto w = rect.getWidth();

121 auto center = rect.getBottom() - offset;

123

126 for (size_t i = 1; i < numSamples; ++i)
127 g.drawLine({ juce::jmap(SampleType(i - 1), SampleType(0), SampleType(numSamples - 1), SampleType(right - w), SampleType(right)),
128 center - gain * data[i - 1],
129 juce::jmap(SampleType(i), SampleType(0), SampleType(numSamples - 1), SampleType(right - w), SampleType(right)),
